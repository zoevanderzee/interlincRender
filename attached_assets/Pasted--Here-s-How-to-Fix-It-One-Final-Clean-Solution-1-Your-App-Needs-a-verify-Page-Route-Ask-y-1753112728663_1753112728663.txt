import { useEffect, useState } from "react";
import { useRouter } from "next/router";
import { getAuth, applyActionCode } from "firebase/auth";

export default function VerifyEmail() {
  const router = useRouter();
  const { oobCode, mode } = router.query;
  const [status, setStatus] = useState("Verifying your email...");

  useEffect(() => {
    if (mode === "verifyEmail" && typeof oobCode === "string") {
      const auth = getAuth();
      applyActionCode(auth, oobCode)
        .then(() => {
          // Sync to backend
          fetch("/api/sync-email-verification", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ email: auth.currentUser?.email })
          });
          setStatus("‚úÖ Email verified successfully. You can now log in.");
          setTimeout(() => router.push("/login"), 3000);
        })
        .catch(() => {
          setStatus("‚ùå Verification failed. This link may be expired or already used.");
        });
    }
  }, [oobCode, mode]);

  return <div>{status}</div>;
}
üß† Why You Need This
This is what actually completes the verification in Firebase. Without calling applyActionCode(), the email is not truly verified.
